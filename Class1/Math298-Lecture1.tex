\section{Finite Precision}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\frame{\frametitle{Finite Precision}

\begin{itemize}
\item Difference between infinite precision and computer arithmetic
\item You will always make an error when approximating a mathematical object on a computer
\item One of the major types of errors is discretization(truncation), which is the difference between the solution of the discrete problem and the exact solution of the mathematical object
\item Example: Suppose you have a function defined by an infinite series and you ``truncate" it at some point, e.g.

\begin{eqnarray*}
\exp(x) &=& 1 + x + \frac{x^2}{2!} +  \frac{x^3}{3!} + \ldots \\
&\approx& 1 + x
\end{eqnarray*}

\end{itemize}

} % end frame

\section{Conditioning}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\frame{\frametitle{Conditioning}

\begin{itemize}
\item Suppose we would like to know how perturbing $x$ will affect the value of $y = f(x)$
\item Define 
\begin{equation*}
\kappa_f(x) = \frac{| x| \cdot | f^{\prime}(x)|}{| f(x)|}
\end{equation*}
\item Then
\begin{equation*}
\frac{| \hat{y} - y |}{|y|} \approx \kappa_f(x) \cdot \frac{| \hat{x} - x |}{|x|}
\end{equation*}
\item Fact: $-\log_{10} ( $relative error$) \approx $ \# decimal digits to which two numbers agree! 
\end{itemize}
{\color{red} Rule of thumb:}You will lose approximately $\log_{10} (\kappa_f(x))$ decimal digits

} % end frame

\section{Stability}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\frame{\frametitle{Stability}
\begin{itemize}
\item Informal definition of stability.  An algorithm for computing $f(x)$ is {\it stable} if it returns a $\hat{y}$ that satisfies:
\begin{equation*}
\frac{| \hat{y} - y |}{|y|} \approx \kappa_f(x) \cdot \frac{| \hat{x} - x |}{|x|}
\end{equation*}

\item Question:  What do you think of the following algorithm for computing $\exp(x)$? \pause
\begin{equation*}
\exp(x) = 1 + x + \frac{x^2}{2!} +  \frac{x^3}{3!} + \ldots 
\end{equation*} \pause

\item Discussion

\end{itemize}

\section{Summary}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\frame{\frametitle{Summary}

\begin{itemize}
\item Beware catastrophic cancellation - know limits of precision
\item Ask about conditioning of a problem
\item Choose stable algorithms so as to not introduce any more loss of precision than is necessary
\item {\color{red} Conditioning} is fundamentally a characteristic of the problem, while {\color{red} stability} is related to algorithms.
\end{itemize}
