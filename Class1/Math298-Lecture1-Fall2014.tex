\section{Finite Precision}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\frame{\frametitle{Finite Precision}

\begin{itemize}
\item Difference between infinite precision and computer arithmetic
\item You will almost always make an error when approximating a mathematical object on a computer
\item One of the major types of errors is discretization(truncation), which is the difference between the solution of the discrete problem and the exact solution (representation) of the mathematical object
\end{itemize}
} % end frame

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\frame{\frametitle{Finite Precision}

\begin{example}
Suppose you have a function defined by an infinite series and you ``truncate" it at some point, e.g.
\begin{eqnarray*}
\exp(x) &=& 1 + x + \frac{x^2}{2!} +  \frac{x^3}{3!} + \ldots \\
&\approx& 1 + x
\end{eqnarray*}
\end{example}

\begin{block}{Discussion:}
How many terms in the series should we use? Are all values of $x$ the same?
\end{block}
} % end frame

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\frame{\frametitle{Finite Precision}

\begin{example}
\begin{equation*}
f^{\prime} (x) = \lim_{h \to 0} \frac{f(x+h) - f(x)}{h} 
\end{equation*}
\end{example}

\begin{block}{Discussion:}
How do we pick $h$ to get the best approximation? 
What value of h should we choose?
\end{block}
} % end frame

\frame{\frametitle{Finite Precision}

{\Large Important Lesson:\\
\vspace{.25in} Don't solve an approximate problem too exactly!}

\vspace{.5in}

\begin{block}{Rule of Thumb}
To approximate 
\begin{equation*} f^{\prime} (x) = \lim_{h \to 0} \frac{f(x+h) - f(x)}{h} \end{equation*}
choose $h$ to perturb about half of the digits in $x$.
\end{block}



} % end frame

\section{Conditioning}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\frame{\frametitle{Conditioning}

\begin{itemize}
\item Suppose we would like to know how perturbing $x$ will affect the value of $y = f(x)$
\begin{definition} [Condition Number]
\begin{equation*}
\kappa_f(x) = \frac{| x| \cdot | f^{\prime}(x)|}{| f(x)|}
\end{equation*}
\end{definition}
\item Then
\begin{equation*}
\frac{| \hat{y} - y |}{|y|} \approx \kappa_f(x) \cdot \frac{| \hat{x} - x |}{|x|}
\end{equation*}
\item Fact: \# decimal digits to which two numbers agree $ \approx -\log_{10} ( $relative error$) $ 
\end{itemize}
\begin{block}{Rule of Thumb}
You will lose approximately $\log_{10} (\kappa_f(x))$ decimal digits when computing anything
\end{block}

} % end frame

\section{Stability}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\frame{\frametitle{Stability}
\begin{itemize}
\item Informal definition of stability.  An algorithm for computing $\hat{y} = f(x)$ is {\it \bf stable} if it returns a $\hat{y}$ that satisfies:
\begin{equation*}
\frac{| \hat{y} - y |}{|y|} \approx \kappa_f(x) \cdot \frac{| \hat{x} - x |}{|x|} .
\end{equation*}

\begin{block}{Discussion:}
What do you think of the following algorithm for computing $\exp(x)$?
\begin{equation*}
\exp(x) = 1 + x + \frac{x^2}{2!} +  \frac{x^3}{3!} + \ldots 
\end{equation*}
\end{block}


\end{itemize}

} % end frame


\section{Summary}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\frame{\frametitle{Summary}

\begin{itemize}
\item Beware catastrophic cancellation - know limits of precision
\item Learn about the conditioning of a problem
\item Choose stable algorithms so as to not introduce any more loss of precision than is necessary
\item Always remember
\begin{itemize}
	\item {\color{red} Conditioning} is fundamentally a characteristic of the problem while
	\item {\color{red} Stability} is related to algorithms
	\end{itemize}
\end{itemize}
